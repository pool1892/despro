\section{34}
\begin{verbatim}
Let towerCat n i j k as = tower (n i j k )++ as

towerCat 0 i j k as 
= tower 0 i j k ++ as 
= []++ as
= as

towerCat n i j k as 
= tower n i j k ++ as
= tower (n-1) i k j ++ [(i,j)] ++ tower (n-1) k j i ++ as
= towerCat (n-1) i k j ([(i,j)]: towerCat (n-1) k j i as)
\end{verbatim}
